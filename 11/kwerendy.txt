1. SELECT filmy.id_filmu, filmy.tytul, filmy.rok_produkcji, filmy.czas_trwania FROM filmy WHERE filmy.czas_trwania >= 100;
2. SELECT aktorzy.imie, aktorzy.nazwisko, aktorzy.plik_awatara FROM aktorzy WHERE aktorzy.id_aktora =3;
3. SELECT * FROM aktorzy ORDER BY aktorzy.nazwisko, aktorzy.imie ASC;
4. SELECT filmy.id_filmu, filmy.tytul, filmy.rok_produkcji FROM filmy JOIN filmy_aktorzy 
   ON filmy.id_filmu = filmy_aktorzy.id_filmu WHERE filmy_aktorzy.id_aktora = 3;
5. SELECT filmy_aktorzy.id_aktora, COUNT(filmy.tytul) AS "liczba filmów" FROM filmy 
   JOIN filmy_aktorzy ON filmy.id_filmu = filmy_aktorzy.id_filmu GROUP BY filmy_aktorzy.id_aktora;
