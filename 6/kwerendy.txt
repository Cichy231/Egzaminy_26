1. SELECT choroby.nazwa FROM choroby WHERE choroby.zakazna = "T" ORDER BY choroby.nazwa ASC;
2. SELECT choroby.id, choroby.nazwa FROM choroby;
3. SELECT objawy.nazwa FROM objawy JOIN choroby_objawy ON objawy.id = choroby_objawy.id_objawy 
   JOIN choroby ON choroby.id = choroby_objawy.id_choroby WHERE choroby.id = 5;
4. SELECT choroby.nazwa, COUNT(choroby_objawy.id_objawy) FROM choroby 
   JOIN choroby_objawy ON choroby.id = choroby_objawy.id_choroby WHERE choroby.zakazna = "T" GROUP BY choroby.nazwa;
