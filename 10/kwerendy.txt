2. SELECT p.nazwa, r.rodzaj FROM potrawy p JOIN rodzaje r ON p.idRodzaje = r.idRodzaje WHERE p.idPotrawy= 7;
3. SELECT p.nazwa, a.alergen FROM potrawy p JOIN lista_alergenow lis ON p.idPotrawy = lis.idPotrawy 
   JOIN alergeny a ON a.idAlergeny = lis.idAlergeny WHERE p.idPotrawy = 7;
